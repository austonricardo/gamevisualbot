function avaliaChar(char as Character, expressao as string) as boolean
	expressao = replace(expressao, "char.cp", char.cp)
	expressao = replace(expressao, "char.hp", char.hp)
	expressao = replace(expressao, "char.mp", char.mp)
	avaliaChar=avalia(expressao)
end function

function avalia(expressao as string) as boolean
	dim pos as integer
	pos =  instr(expressao, " ou ")
	if pos > 0 then
		avalia = avalia(left(expressao,pos)) or avalia(mid(expressao,pos + 4))	
	else
		pos =  instr(expressao, " e ")
		if pos > 0 then
			avalia = avalia(left(expressao,pos)) and avalia(mid(expressao,pos + 3))
		else
			'Avalia expressão
			pos =  instr(expressao, "<")
			
		end if
	end if

	err.raise "avaliação da expressão não foi possivel:[" & expressao & "]"
end function
